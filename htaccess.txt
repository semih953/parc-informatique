# .htaccess - Protection pour serveur Apache (production)
# Équivalent du router.php pour quand le site sera hébergé sur un vrai serveur

# Désactiver le listing des dossiers
Options -Indexes

# Bloquer tous les fichiers JSON (users.json, données)
<FilesMatch "\.(json|log|bak|old|sql)$">
    Require all denied
</FilesMatch>

# Bloquer le dossier data/
RedirectMatch 403 ^/data(/.*)?$

# Bloquer les fichiers cachés (.git, .gitignore, .env...)
RedirectMatch 403 /\..*$

# Empêcher l'exécution de PHP dans uploads/
<If "%{REQUEST_URI} =~ m#^/uploads/#">
    <FilesMatch "\.ph(p[0-9]?|tml|ar)$">
        Require all denied
    </FilesMatch>
</If>

# En-têtes de sécurité (doublon de security.php, par précaution)
<IfModule mod_headers.c>
    Header always set X-Frame-Options "DENY"
    Header always set X-Content-Type-Options "nosniff"
    Header always set Referrer-Policy "same-origin"
</IfModule>
